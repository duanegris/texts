\overlays{4}{
\begin{slide}{Notion de projet}

\vspace{1cm}
Ensemble d'activités :
\vspace{.2cm}
\begin{itemstep}
\item appartenant à différentes phases  
\item ayant un objectif commun
\item permettant la satisfaction d'un besoin identifié 
\item nécessitant des équipes de spécialistes aux compétences variées 
\item sur lesquelles s'exercent trois types de contraintes : coûts, délais, qualité 
\end{itemstep}
\vfill
\end{slide}}
%*****************************************************************
\overlays{4}{
\begin{slide}{Caractéristiques d'un projet}

\vspace{1cm}
\begin{itemstep}
\item une action unique et ponctuelle, non répétitive  
\vspace{.2cm}
\item limité dans le temps : dates de début et de fin
\vspace{.2cm}
\item une démarche spécifique : atteindre l'objectif en maîtrisant
      la qualité du produit fini, les coûts et les délais grâce à des
	étapes et des jalons
	
\vspace{.2cm}
\item mobilise des compétences multiples et complémentaires 
\end{itemstep}
\vfill
\end{slide}
}

\overlays{2}{
\begin{slide}{Lancement d'un projet}

\vspace{.5cm}
Avant acceptation ou lancement, se poser des questions :\\ 

\vspace{.5cm}
Toute difficulté identifiée devra faire l'objet 
d'un dialogue approfondi avec le demandeur pour

\vspace{.5cm}
\begin{itemize}
\item soit annuler, infléchir ou différer le projet
\item soit négocier des moyens de réussite à hauteur des enjeux et des conditions de
réussite identifiées.
\end{itemize}

\FromSlide{2}
$\Rightarrow$ Une \underline{évaluation} nécessaire
\end{slide}}

%*****************************************************************
\overlays{4}{
\begin{slide}{Evaluer le projet}
\'{E}valuer sous différents angles :
\begin{itemstep}
\item Evaluer par le résultat attendu
\item Evaluer par la pertinence de la demande 
\item Evaluer par la cohérence du projet dans le contexte de l'organisation
\item  Evaluer par la conduite de projet
\end{itemstep}
\vfill
\end{slide}}
%*****************************************************************
\begin{slide}{Evaluer le projet (2)}

\begin{itemize}
\item Evaluer par le résultat attendu et les enjeux 
\begin{small}  
    \begin{itemize}
    \item générateur de résultats économiques ?
    \item initiateur de changements dans les structures et comportements ? 
    \end{itemize}
\end{small}  

\item
Evaluer par la pertinence de la demande : est t-elle mûre ?
\begin{small}  
    \begin{itemize}
    \item identifier le demandeur (initiateur, décideur, destinataire)
    \item cerner la demande : demande énoncée clairement, nature de la demande, cadre de la demande, les délais 
    \end{itemize}
\end{small}  
\end{itemize}
\end{slide}


\begin{slide}{Evaluer le projet (3)}
\begin{itemize}
\item
Evaluer par la cohérence du projet dans le contexte de l'organisation
    \begin{itemize}
    \item Les résultats escomptés sont ils en accord avec la stratégie 
          de l'entreprise ? 
    \item Le projet s'inscrit-il dans la planification générale de l'entreprise  ?
    \item Comment se positionne le projet vis-à-vis d'autres projets ou actions ? (antinomies, synergies, compétition )
    \end{itemize}
\end{itemize}
\vfill
\end{slide}


\begin{slide}{Evaluer le projet (4)}
\begin{itemize}
\item Evaluer par la conduite de projet : les risques d'aléas
    \begin{itemize}
    \item garanties de progression et d'achèvement ? 
    \item programme des étapes et décisions intermédiaires connus ?
    \item les indicateurs de bonne fin sont ils précisés ?
    \end{itemize}
	
\end{itemize}
\vfill
\end{slide}

\begin{slide}{Les causes d'échecs (1)}

\vspace{1cm}

\underline{Côté fournisseur}
\begin{itemize}
\item syndrome du cahier des charges
\item mise en \oe uvre de moyens inadaptés
\item contraintes de délais, charges et coûts
\item démotivation de l'équipe
\item absence d'outils
\end{itemize}

\vspace{2cm}
\begin{center}
$\Rightarrow$ mène à la perte de contrôle du projet
\end{center}
\vfill
\end{slide}
 
%*****************************************************************
\begin{slide}{Les causes d'échecs (2)}

\vspace{1cm}

\underline{Côté utilisateur}
\begin{itemize}
\item l'incapacité à dialoguer entre partenaires
\item la mauvaise définition des objectifs
\item l'absence d'implication des utilisateurs
\item le manque de qualité du produit livré
\item l'absence de calcul des risques
\end{itemize}

\vspace{2cm}
\begin{center}
$\Rightarrow$ mène à un résultat décevant ou à l'échec
\end{center}
\vfill
\end{slide}
%*****************************************************************
\begin{slide}{La perte de contrôle}

Quelques symptômes :
\vspace{1cm}
\begin{itemize}
\item perte de contrôle sur certains responsables
\item on ne peut plus s'engager sur la date de livraison 
\item la qualité du produit en développement est incertaine
\item non-détection à temps les écarts de délai, de coûts ou de
conformité aux spécifications 
\end{itemize} 

\vfill
\end{slide}

\overlays{2}{
\begin{slide}{Motiver l'équipe de projet}

Chaque acteur s'engagera d'autant plus que :
 
\begin{itemize}
\item le résultat de son action est visible
\item le résultat envisagé est désiré
\item sa confiance dans sa capacité à agir est grande
\end{itemize} 

\FromSlide{2}

Provoquer ces trois facteurs :
\begin{itemize}
\item s'accorder sur le chemin à parcourir
\item assurer la continuité du processus
\item faire adhérer les équipes
\item agir par la hiérarchie
\item investir en ressources humaines, temps, argent
\end{itemize} 

\vfill
\end{slide}}
