\section{Réseau}

\begin{frame}{Adresses IPv4 : Classe}

			Organisation par classes
			\begin{itemize}
			\item Défini la longueur des parties réseau et hôte
			\item On achete une \textit{plage} d'adresse définie par la partie hôte
			\end{itemize}

		\begin{block}<2-> {Classes}
			\begin{description}
			\item [Classe A] : 1 octet réseau, commence par \hfill \texttt{0\ \ \ \ldots}
			\item [Classe B] : 2 octets réseau, commence par \hfill \texttt{10\ \ \ldots}
			\item [Classe C] : 3 octets réseau, commence par \hfill \texttt{110\ \ldots}
			\item [Classe D] : multicast, commence par \hfill \texttt{1110\ldots}
			\item [Classe E] : expérimental, le reste
			\end{description}
		\end{block}		
	
\end{frame}

\section{Adresses particulières}

% ----------------------------------------------------------------------
\begin{frame}{Adresses particulières}

		\begin{block}<+-> {Par masque (ex classe A sans sous-réseau)}
			\begin{itemize}
			\item \texttt{\ \ X.\textcolor{blue}{\ \ 0.\ \ 0.\ \ 0}} : adresse réseau
			\item \texttt{\ \ X.\textcolor{blue}{255.255.255}}  : adresse (\textit{broadcast})
			\item \texttt{\textcolor{red}{\ \ 0}.\ \ X.\ \ X.\ \ X} : adresse hôte dans le sous-réseau courant
			\item \texttt{\textcolor{red}{255}.\ \ X.\ \ X.\ \ X} : adresse (\textit{multicast})
			\item \texttt{\textcolor{red}{255}.\textcolor{blue}{255.255.255}} : adresse (\textit{broadcast universel}) 			
			\end{itemize}		
		\end{block}
		
		\begin{block}<+-> {Adresses privées}
			\begin{itemize}
			\item L'adresse de classe A 10.0.0.0 
			\item Les adresses réseau de classe B 172.16.0.0 à 172.31.0.0 
			\item Les adresses réseau de classe C 192.168.0.0 à 192.168.255.0 
			\end{itemize}
		\end{block}

		\begin{block}<+-> {Boucle locale}
			\begin{itemize}
			\item 127.0.0.1 = \textit{localhost}
			\end{itemize}		
		\end{block}
		
\end{frame}
