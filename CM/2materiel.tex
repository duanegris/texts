	\begin{block}<+-> {Répéteur (\textit{repeater})}
		\begin{itemize}
		\item Regénére le signal entre deux câbles
		\item Pour dépasser les limites d'atténuation
		\item Répète sur un port ce qu'il entend sur l'autre
		\item Couche OSI 1
		\end{itemize}	
	\end{block}

\subsection{Matériel interne}
% ----------------------------------------------------------------------
\begin{frame}

\centering
\includegraphics[width=\textwidth]{\resdir/pont.jpg}

	\begin{block}<+-> {Pont (\textit{bridge})}
		\begin{itemize}
		\item Relie des réseau travaillant avec le même protocole
		\item = Filtre les trames à destination du réseau de l'autre côté
		\item Permet de segmenter le réseau (réduction du trafic)
		\item Couche OSI 1 et 2 (adresse MAC)
		\end{itemize}	
	\end{block}
	
\end{frame}
% ----------------------------------------------------------------------

% ----------------------------------------------------------------------
\begin{frame}

\centering
\includegraphics[height=0.5\textheight]{\resdir/routeur.jpg}

	\begin{block}<+-> {Routeur (\textit{router})}
		\begin{itemize}
		\item Route les paquets entre les différents réseaux
		\item Interne à l'internet
		\item Détermine le chemin à emprunter (table de routage)
		\item Implémente un algorithme de routage
		\end{itemize}	
	\end{block}
	
\end{frame}
% ----------------------------------------------------------------------


% ----------------------------------------------------------------------
\begin{frame}

\centering
\includegraphics[width=0.32\textwidth]{\resdir/neufbox.jpg}\hfill
\includegraphics[width=0.32\textwidth]{\resdir/freebox.jpg}\hfill
\includegraphics[width=0.32\textwidth]{\resdir/livebox.jpg}

	\begin{block}<+-> {Passerelle (\textit{gateway})}
		\begin{itemize}
		\item Pont entre deux réseaux de protocole différents
		\item Matériel et logiciel
		\item Exemple : modem ou tunnel (VPN)
		\end{itemize}	
	\end{block}
	
\end{frame}
% ----------------------------------------------------------------------


% ----------------------------------------------------------------------
\begin{frame}

\centering
\includegraphics[height=\imgheight]{\resdir/concentrateur}

	\begin{block}<+-> {Concentrateur (\textit{hub})}
		\begin{itemize}
		\item Concentre le trafic de plusieurs hôtes
		\item Rediffusion du trafic arrivant sur chaque port sur tous les autres
		\item \'Egalement : répéteur
		\item 4/8/16/32 ports
		\item Couche OSI 1 
		\item Actif (alimenté) ou passif
		\item Connectable en cascade (port uplink/câble croisé)
		\end{itemize}	
	\end{block}
	
\end{frame}
% ----------------------------------------------------------------------


% ----------------------------------------------------------------------
\begin{frame}

\centering
\includegraphics[width=\textwidth]{\resdir/commutateur.jpg}

	\begin{block}<+-> {Commutateur (\textit{switch})}
		\begin{itemize}
		\item Pont multiports
		\item Concentre et redirige les données vers la bonne destination
		\item 4-96 ports		
		\item Couche OSI 1 et 2
		\item \'Egalement : commutateur de niveau 7
		\end{itemize}	
	\end{block}
	
\end{frame}
% ----------------------------------------------------------------------


% ----------------------------------------------------------------------
\begin{frame}

\centering
\includegraphics[height=\imgheight]{\resdir/parefeu.jpg}

	\begin{block}<+-> {Pare-feu (\textit{firewall})}
		\begin{itemize}
		\item Filtre les communications (sécurité)
		\item Matériel ou logiciel
		\item Réseau (par port/protocole) ou applicatif (par application/port/protocole)
		\item Analyse plus ou moins profonde des communications
		\item Gestion dynamique des ports
		\item Permet aussi de faire des translations (NAT/PAT)
		\end{itemize}	
	\end{block}
	
\end{frame}
