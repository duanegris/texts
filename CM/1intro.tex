\subsection{Historique}
% ----------------------------------------------------------------------
\begin{frame}{Historique}
	
	\begin{itemize}
		\item<+-> 1968 : Arpanet (Advanced Research Projects Agency), DOD américain, mel entre centres de recherche, protocole X25
		\item<+-> 1972 : protocole TCP/IP, résister à une attaque nucléaire
		\item<+-> 1979 : mise en domaine public, récupération par les université (Duke University, Caroline du Nord), USENET (news)
		\item<+-> 1982 : accès gratuit au réseau
		\item<+-> 1983 : arrivée en Europe, CNAM en France, réseau de réseaux
		\item<+-> 1984 : ARPANET devient MILNET + INTERNET par la NFS
		\item<+-> 1986 : connexion aux lignes publiques
		\item<+-> 1991 : Apparition des FAI
		\item<+-> 1993 : WWW, 1ier navigateur au CERN, HTML/hyperliens
		\item<+-> 1995 : 2 millions d'ordinateurs, 30 millions d'utilisateurs
		\item<+-> 2008 : 15,5 millions de foyers haut-débit en France, +1milliard d'internautes dans le monde
	\end{itemize}

\end{frame}

\begin{frame}
