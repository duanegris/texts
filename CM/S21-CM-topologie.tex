\section{DNS}
% ----------------------------------------------------------------------
\subsection{Présentation}
% ----------------------------------------------------------------------
\begin{frame}
	\begin{block}<+-> {DNS : Domain Name System}
		\begin{itemize}
		\item Fournit l'association : \textbf{nom de domaine} <-> \textbf{IP} 
		\item Demandé par Jon Postel
		\item Implémenté par Paul Mockapetris en 1983
		\item Pour faciliter la mémorisation humaine des machines
		\item Centaines de milliers de \textbf{serveurs DNS}
		\end{itemize}	
	\end{block}
	
	\begin{block}<+-> {Fonctionnalités}	
		\begin{itemize}
		\item Résolution : \textbf{nom de domaine} -> \textbf{IP} 
		\item Résolution inversée : \textbf{IP} -> \textbf{nom de domaine}
		\item Tourniquet : \textbf{nom de domaine} -> plusieurs \textbf{IP} 
			\begin{itemize}
			\item Données à tour de rôle
			\item \'Equilibrage des charges
			\end{itemize}			
		\end{itemize}	
	\end{block}
	
\end{frame}
% ----------------------------------------------------------------------

% ----------------------------------------------------------------------
\begin{frame}

	\begin{block}<+-> {FQDN}
		\begin{itemize}
		\item FQDN : Fully Qualified Domain Name
		\item ou Nom de Domaine Pleinement Qualifié
		\item Format : hôte.domaine.tld
			\begin{itemize}
			\item hôte = nom de la machine
			\item domaine = domaine d'appartenance de la machine
			\item tld = Top Level Domain \textit{.fr .com .org .net}\ldots
			\end{itemize}
		\item Exemple : \textit{iut-info.unistra.fr}
		\end{itemize}	
	\end{block}
	
	\begin{block}<+-> {Inversion}
		\begin{itemize}
		\item terme plus global à droite dans les noms de domaine
			\begin{itemize}
			\item iut-info.unistra.\textbf{fr}
			\end{itemize}
		\item terme plus global à gauche dans les adresses IP
			\begin{itemize}
			\item \textbf{91}.198.174.2
			\end{itemize}			
		\item inversion pour la cohérence :
			\begin{itemize}
			\item 91.198.174.2 devient  2.174.198.91.in-addr.arpa
			\end{itemize}
		\end{itemize}	
	\end{block}
	
		
\end{frame}
% ----------------------------------------------------------------------

% ----------------------------------------------------------------------
\subsection{Topologie}
% ----------------------------------------------------------------------
\begin{frame}

	\begin{block}<+-> {Fonctionnement}
		\begin{itemize}
		\item FQDN = séquence de n\oe uds, de la machine au tld
		\item Correspond à une chaine de serveurs DNS
		\item Organisés en hierarchie
		\end{itemize}
	\end{block}


		\includegraphics<+->[width=\textwidth]{res/DNS.png}
	
\end{frame}
% ----------------------------------------------------------------------


% ----------------------------------------------------------------------
\begin{frame}
	
	
	
	\begin{exampleblock}<+-> {Exemple : pour résoudre \texttt{iut-info.unistra.fr}}
		\begin{itemize}
		\item le serveur DNS local contacte le serveur DNS \texttt{racine}
		\item qui renvoie la requête au serveur DNS \texttt{.fr}
		\item qui renvoie la requête au serveur DNS \texttt{.unistra.fr}
		\item qui renvoie l'IP de la machine \texttt{iut-info}			
		\end{itemize}
	\end{exampleblock}
	
	\begin{alertblock}<+-> {Accéléré par des \textbf{caches}}
		\begin{itemize}
		\item Sur les serveurs DNS et sur les machines (fichier \textit{host})
		\end{itemize}
	\end{alertblock}
		
\end{frame}
% ----------------------------------------------------------------------




% ----------------------------------------------------------------------
\begin{frame}{Sécurité}

	\begin{block}<+-> {Problème}
		\begin{itemize}
		\item Système névralgique d'Internet
		\item Mais sécurité non prévue à la base
		
		\item Objectifs
			\begin{itemize}
			\item Rendre Internet inutilisable, Fishing
			\end{itemize}
			
		\item Ne pas faire confiance
			\begin{itemize}
			\item Certificats, SSH (fichier \textit{known\_hosts})
			\end{itemize}
		\end{itemize}
	\end{block}	
	
	\begin{block}<+-> {Attaques}
		\begin{itemize}
		\item l'interception du paquet (requête ou réponse) et émission d'un autre paquet à sa place
		\item la fabrication d'une réponse 
		\item la trahison par un serveur ou corruption de données
		\item l'empoisonnement du cache DNS 
		\item le déni de service
		\end{itemize}
	\end{block}	
		
\end{frame}
