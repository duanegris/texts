\section{Modèle OSI}
% ----------------------------------------------------------------------
\begin{frame}
	\begin{columns}
	
	\column{0.5\textwidth}
		\includegraphics[width=\textwidth]{res/osi_com}
	
	\column{0.5\textwidth}
		\begin{block}<+-> {Modèle OSI}
			\begin{itemize}
				\item Séparation en couches
				\item Chacune rend un service à celle juste au dessus
				\item Les données doivent traverser toutes les couches
				\item Avantages :
				\begin{itemize}
					\item Réduit la complexité
					\item Uniformise les interfaces
					\item Assure l'interopérabilité
					\item Facilite la conception modulaire
					\item Accélère l'évolution
				\end{itemize}
			\end{itemize}
		\end{block}
		
	\end{columns}


\end{frame}
