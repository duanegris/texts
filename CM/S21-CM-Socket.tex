	\begin{block}<+->{Architectures applicative}
		\begin{itemize}
		\item Gestion des communications au sein d'une application
		\item Choix :
			\begin{itemize}
			\item Quoi échanger ? (Données totales ou mises à jour)
			\item Comment échanger ? (protocole)
			\end{itemize}
		\item Dépend de :
			\begin{itemize}
			\item Objectifs (Nature de l'application)
			\item Données (Taille, Fréquence des mise à jour)
			\item Contraintes (Performance, Fiabilité)
			\end{itemize}
		\item En général 
			\begin{itemize}
			\item plus de performance = moins de fiabilité 
			\item et vice versa
			\end{itemize}		
		\end{itemize}		
	\end{block}
	
	\begin{alertblock}<+->{Deux modes de communications principaux}
		\begin{itemize}
		\item Déconnecté (UDP) et Connecté (TCP)
		\end{itemize}		
	\end{alertblock}

\begin{frame}

	\begin{block}<+->{Notion de port}
		\begin{itemize}
		\item Nombre sur 16 bits (0 à 65535)
		\item Permet de multiplexer les liaisons applicatives dans une liaison physique
		\item Trois catégories :
			\begin{itemize}
			\item[0-1023] : ports connus (\textit{Well Known Ports}) 
			\item[1024-49151] : ports enregistrés (\textit{Registered Ports}) 
			\item[49152-65535] : Ports dynamiques et/ou privés (\textit{Dynamic and/or Private Ports}) 
			\end{itemize}
		\item Gérés par l'IANA (Internet Assigned Numbers Authority)
			\begin{itemize}
			\item \url{http://www.iana.org/assignments/port-numbers}
			\end{itemize}
		\end{itemize}		
	\end{block}
	
	\begin{alertblock}<+->{Exemple}
		\begin{itemize}
		\item Protocole FTP
		\item Construit au dessus de TCP
		\end{itemize}		
	\end{alertblock}	
\end{frame}

\section{Session UDP}
% ----------------------------------------------------------------------
\begin{frame}
	\begin{block}<+->{Protocole UDP}
		\begin{itemize}
		\item \textit{User Datagram Protocol}
		\item Construit par dessus IP
		\item Défini par la RFC768 
			\begin{itemize}
			\item \url{http://tools.ietf.org/html/rfc768}
			\end{itemize}
		\item Intègre la notion de port
		\item En mode déconnecté
			\begin{itemize}
			\item performant (peu de surcharge)
			\item peu fiable (perte, intégrité et ordonnancement des paquets)
			\end{itemize}
		\end{itemize}
	\end{block}
\end{frame}

\section{Session TCP}
% ----------------------------------------------------------------------
\begin{frame}
	\begin{block}<+->{Protocole TCP}
		\begin{itemize}
		\item \textit{Transmission Control Protocol }
		\item Construit par dessus IP
		\item Défini par la RFC793
			\begin{itemize}
			\item \url{http://tools.ietf.org/html/rfc793}
			\end{itemize}
		\item Intègre la notion de port
		\item En mode connecté
			\begin{itemize}
			\item peu performant (beaucoup de surcharge)
			\item fiable (chaque paquet est acquitté et checksumé)
			\end{itemize}
		\end{itemize}
	\end{block}		
\end{frame}
